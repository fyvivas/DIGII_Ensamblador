MAIN1
	MOVF DISP_COUNTER
	MOVWF bin ; 0x20
	CALL Binary2BCD ; tens = 3 unit = 2 
	CALL DISP_UPDATE_UNIT ; 2
	CALL DELAY_20MS
	CALL DISP_UPDATE_TENS; 3
	CALL DELAY_20MS
	DECFSZ DISP_FREQ
	GOTO MAIN1
	DECF DISP_COUNTER,F ;0x1f = 31 decimal
	movlw 20h ; Configura nuestro registro w con 02h
	movwf DISP_FREQ
	GOTO MAIN1
	
DISP_UPDATE_UNIT
	BCF PORTA,1 ;TURN OFF DIGIT 2.
	MOVF Units
	CALL SEVENSEG_LOOKUP
	MOVWF PORTB ;PUT DATA ON PORTB.
	BSF PORTA,0 ;TURN ON DIGIT 1.
	RETURN
	
DISP_UPDATE_TENS
	BCF PORTA,0 ;TURN OFF DIGIT 1.
	MOVF Tens
	CALL SEVENSEG_LOOKUP
	MOVWF PORTB ;PUT DATA ON PORTB.
	BSF PORTA,1 ;TURN ON DIGIT 2.
	RETURN
